# 🔑 M-Pesa Integration Flow for PesaPlan

### 1. **Authentication**

* Generate a token using Safaricom’s OAuth endpoint.
* Cache it (Redis) to avoid regenerating each time.

### 2. **STK Push (C2B)**

* Used when **user is paying** (e.g., topping up wallet, or direct standing order payment).
* User approves on phone (PIN).

### 3. **B2C / B2B Payments**

* Used when **app sends money out** (e.g., sending to recipient).
* Needs higher compliance (float account).

# 📝 Django: M-Pesa Service File

Create a helper for interacting with Daraja API:

```python
# mpesa_service.py
import requests
from requests.auth import HTTPBasicAuth
import base64
from datetime import datetime
from django.conf import settings


class MpesaClient:
    def __init__(self):
        self.consumer_key = settings.MPESA_CONSUMER_KEY
        self.consumer_secret = settings.MPESA_CONSUMER_SECRET
        self.shortcode = settings.MPESA_SHORTCODE
        self.passkey = settings.MPESA_PASSKEY
        self.base_url = "https://sandbox.safaricom.co.ke" if settings.DEBUG else "https://api.safaricom.co.ke"

    def get_access_token(self):
        url = f"{self.base_url}/oauth/v1/generate?grant_type=client_credentials"
        response = requests.get(url, auth=HTTPBasicAuth(self.consumer_key, self.consumer_secret))
        return response.json()["access_token"]

    def stk_push(self, phone_number, amount, account_reference="PesaPlan", transaction_desc="Standing Order Payment"):
        access_token = self.get_access_token()
        timestamp = datetime.now().strftime("%Y%m%d%H%M%S")

        password = base64.b64encode(
            (self.shortcode + self.passkey + timestamp).encode("utf-8")
        ).decode("utf-8")

        payload = {
            "BusinessShortCode": self.shortcode,
            "Password": password,
            "Timestamp": timestamp,
            "TransactionType": "CustomerPayBillOnline",
            "Amount": str(amount),
            "PartyA": phone_number,  # customer phone
            "PartyB": self.shortcode,
            "PhoneNumber": phone_number,
            "CallBackURL": settings.MPESA_CALLBACK_URL,
            "AccountReference": account_reference,
            "TransactionDesc": transaction_desc,
        }

        headers = {"Authorization": f"Bearer {access_token}"}
        url = f"{self.base_url}/mpesa/stkpush/v1/processrequest"
        response = requests.post(url, json=payload, headers=headers)

        return response.json()
```

# 📝 DRF View: Trigger STK Push for Standing Orders

```python
from rest_framework.response import Response
from rest_framework.decorators import action
from rest_framework import status
from .mpesa_service import MpesaClient
from .models import StandingOrder, Transaction


class StandingOrderViewSet(viewsets.ModelViewSet):
    serializer_class = StandingOrderSerializer
    permission_classes = [permissions.IsAuthenticated]

    def get_queryset(self):
        return StandingOrder.objects.filter(user=self.request.user)

    def perform_create(self, serializer):
        standing_order = serializer.save(user=self.request.user)
        
        # Trigger M-Pesa payment immediately for first run
        mpesa = MpesaClient()
        response = mpesa.stk_push(
            phone_number=self.request.user.phone_number,
            amount=standing_order.amount,
            account_reference=f"SO-{standing_order.id}"
        )

        # Save transaction
        Transaction.objects.create(
            user=self.request.user,
            standing_order=standing_order,
            amount=standing_order.amount,
            status="pending"
        )

        return standing_order

    @action(detail=True, methods=["post"])
    def execute(self, request, pk=None):
        """Manually trigger payment for a standing order"""
        standing_order = self.get_object()
        mpesa = MpesaClient()
        response = mpesa.stk_push(
            phone_number=request.user.phone_number,
            amount=standing_order.amount,
            account_reference=f"SO-{standing_order.id}"
        )

        Transaction.objects.create(
            user=request.user,
            standing_order=standing_order,
            amount=standing_order.amount,
            status="pending"
        )

        return Response(response, status=status.HTTP_200_OK)
```

# 📝 Handling Callbacks

M-Pesa sends a callback after payment success/failure → we must capture it:

```python
from rest_framework.views import APIView

class MpesaCallbackView(APIView):
    def post(self, request):
        data = request.data
        result_code = data["Body"]["stkCallback"]["ResultCode"]
        mpesa_receipt = None

        if result_code == 0:
            mpesa_receipt = data["Body"]["stkCallback"]["CallbackMetadata"]["Item"][1]["Value"]
            status = "success"
        else:
            status = "failed"

        # Update last pending transaction
        checkout_id = data["Body"]["stkCallback"]["CheckoutRequestID"]
        transaction = Transaction.objects.filter(status="pending").last()
        if transaction:
            transaction.status = status
            transaction.mpesa_receipt = mpesa_receipt
            transaction.save()

        return Response({"message": "Callback received"}, status=200)
```

Add to **urls.py**:

```python
urlpatterns += [
    path("mpesa/callback/", MpesaCallbackView.as_view(), name="mpesa-callback"),
]
```
